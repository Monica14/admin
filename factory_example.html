The more helpful concept which using factory we can implement is : Encapsulation patterns in AngularJS. The concept of Encapsulation is an important design pattern in most object-oriented programming designs. Determining which properties and methods should be accessible is a nice forcing function because it forces the programmer to think about how his or her could will be consumed. 

            app.factory('factory_exam', function ($rootScope) {
                var displayFact;
            
                var addMSG = function (msg) {
                    displayFact = ' Hi Guest, Welcome to ' + msg;
                    //console.log(displayFact)
                }
                ///////////////// private function ////////////
                function getMSG()
                {
                     alert(displayFact);
                }
                ///////////////// public function ////////////
                return {
                    setMSG: function (msg) {
                        addMSG(msg);
                    },  
                    getMSG: function () {
                        alert(displayFact);
                    },      
                }
            })
