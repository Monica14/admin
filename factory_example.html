app.factory('factory_exam', function ($rootScope) {
                var displayFact;
            
                var addMSG = function (msg) {
                    displayFact = ' Hi Guest, Welcome to ' + msg;
                    //console.log(displayFact)
                }
                return {
                    setMSG: function (msg) {
                        addMSG(msg);
                    },  
                    getMSG: function () {
                        alert(displayFact);
                    },      
                }
            })
