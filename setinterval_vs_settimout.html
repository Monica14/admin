setInterval vs setTimeout

                    
                    for(var i=0;i<5;i++)
                    {
                        setTimeout(()=>{ console.log(i) } ,1000);      
                        console.log("Hello")                   
                    }                        
                
                    
                    for(var i=0;i<5;i++)
                    {
                        setInterval(()=>{ console.log(i) } ,1000);      
                        console.log("Hello")                   
                    }
